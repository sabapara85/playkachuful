━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
 STEP 1 — Create GitHub Account (if you don't have one)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  Go to: https://github.com/signup
  Create a free account with your email.

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
 STEP 2 — Upload your game to GitHub
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  1. Go to: https://github.com/new
  2. Repository name: kachuful
  3. Set to: Public
  4. Click "Create repository"

  Then upload your files:
  5. Click "uploading an existing file"
  6. Drag and drop ALL files from your KACHUFUL folder:
       - server.py
       - requirements.txt
       - render.yaml
       - templates/  (folder)
       - static/     (folder)
  7. Click "Commit changes"

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
 STEP 3 — Create Render Account
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  Go to: https://render.com
  Sign up with your GitHub account (easiest)

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
 STEP 4 — Deploy on Render
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  1. In Render dashboard, click "New +" → "Web Service"
  2. Connect your GitHub account
  3. Select your "kachuful" repository
  4. Render will auto-detect settings from render.yaml
  5. Scroll down, click "Create Web Service"
  6. Wait 3-5 minutes for deployment
  7. You will see your live URL:
       https://kachuful.onrender.com
       (or similar name)

 STEP 5 — Share with friends
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

 ⚠️  FREE TIER LIMITATION
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  Render free tier "sleeps" after 15 minutes of no traffic.
  First person to open the link may wait 30-60 seconds
  for the server to wake up. After that, everyone plays
  normally with no delays.

  FIX: Open the link yourself 2 minutes before friends join.
  That wakes it up and everyone connects instantly.

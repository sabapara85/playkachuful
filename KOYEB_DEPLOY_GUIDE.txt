 BEFORE YOU START — CHECKLIST
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  ✅ GitHub account with kachuful repository
  ✅ Latest files uploaded to GitHub:
       - server.py
       - requirements.txt
       - Procfile          ← NEW FILE (included in this package)
       - render.yaml
       - templates/index.html

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

 STEP 1 — UPLOAD NEW FILES TO GITHUB
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  1. Go to github.com → your kachuful repository
  2. Upload/replace these files:
       a. server.py (root folder)
       b. requirements.txt (root folder)
       c. Procfile (root folder) ← NEW — drag and drop this file
       d. templates/index.html (templates folder)
  3. Commit changes

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

 STEP 2 — CREATE KOYEB ACCOUNT
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  1. Go to: https://www.koyeb.com
  2. Click "Get Started Free"
  3. Click "Sign up with GitHub"
     → This automatically links your GitHub account
  4. Authorize Koyeb to access your repositories
  5. You are now on Koyeb dashboard

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

 STEP 3 — CREATE NEW APP
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  1. Click "Create App" button on dashboard
  2. Select source: "GitHub"
  3. Select repository: "kachuful"
  4. Select branch: "main"

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

 STEP 4 — CONFIGURE BUILD SETTINGS
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  Builder:      Buildpack (should auto-detect Python)
  
  Run command:  gunicorn --worker-class geventwebsocket.gunicorn.workers.GeventWebSocketWorker --workers 1 --bind 0.0.0.0:$PORT server:app
  
  NOTE: If Koyeb detects Procfile automatically, the run command
  is already set — no need to type it manually.

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

 STEP 5 — ENVIRONMENT VARIABLES
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  Click "Add Variable" and add:

  Key:    RENDER
  Value:  true

  (This tells the server to use gevent mode — required)

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

 STEP 6 — INSTANCE TYPE AND APP NAME
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  Instance type:  Free
  App name:       kachuful

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

 STEP 7 — DEPLOY
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  1. Click "Deploy" button
  2. Wait 3-5 minutes — you will see logs scrolling
  3. When you see "Running" — deployment successful!
  4. Koyeb gives you a free URL like:
       https://kachuful-xxxx.koyeb.app

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

 STEP 8 — TEST FIRST (IMPORTANT)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  Before connecting domain, test on Koyeb URL:

  1. Open https://kachuful-xxxx.koyeb.app
  2. Create room → join from another browser
  3. Start game → bid → play a card
  4. Test chat → test music
  5. Test skip player (disconnect one player, host skips)

  If all works → proceed to Step 9
  If issues → contact Claude with the error message

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

 STEP 9 — ADD CUSTOM DOMAIN IN KOYEB
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  1. In Koyeb dashboard → click your kachuful app
  2. Click "Settings" tab
  3. Scroll to "Domains" section
  4. Click "Add Domain"
  5. Type: playkachuful.online
  6. Click Save
  7. Koyeb shows you a CNAME value like:
       kachuful-xxxx.koyeb.app
     NOTE IT DOWN — you need this in next step

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

 STEP 10 — CONNECT DOMAIN IN NAMECHEAP
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  1. Go to namecheap.com → Login
  2. Domain List → playkachuful.online → Manage
  3. Click "Advanced DNS" tab
  4. Delete any existing A or CNAME records for @ and www
  5. Click "Add New Record" and add:

  Record 1:
    Type:  CNAME
    Host:  @
    Value: kachuful-xxxx.koyeb.app (from Step 9)
    TTL:   Automatic

  Record 2:
    Type:  CNAME
    Host:  www
    Value: kachuful-xxxx.koyeb.app (same value)
    TTL:   Automatic

  6. Click green tick ✓ to save each record

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

 STEP 11 — WAIT AND TEST
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  Wait 15-30 minutes for DNS to propagate, then test:

    https://playkachuful.online      ← Should open game
    https://www.playkachuful.online  ← Should also open game

  If it works — YOU ARE LIVE! 🎉

 TROUBLESHOOTING
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  Problem: Build fails on Koyeb
  Fix: Make sure Procfile is uploaded to root of GitHub repo

  Problem: App starts but game doesn't work
  Fix: Check RENDER=true environment variable is set

  Problem: Domain not working after 30 min
  Fix: Double-check CNAME records in Namecheap Advanced DNS

  Problem: SSL error
  Fix: Wait 10 more minutes — Koyeb auto-issues SSL certificate

  Problem: WebSockets not working
  Fix: Check if Koyeb has WebSocket support enabled in settings
